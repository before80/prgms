JDBCпјҲJava Database ConnectivityпјүжҳҜ Java е®ҳж–№жҸҗдҫӣзҡ„ж•°жҚ®еә“и®ҝй—® APIгҖӮе®ғеҮәзҺ°еңЁ JDK зҡ„ java.sql е’Ң javax.sql дёӨдёӘеҢ…йҮҢпјҢжҳҜ Java дёҺж•°жҚ®еә“йҖҡдҝЎзҡ„"еҺҹз”ҹеҚҸи®®"гҖӮ

дҪ еҸҜд»ҘжҠҠ JDBC жғіиұЎжҲҗпјҡдҪ жӢҝзқҖдёҖжң¬еҺҡеҺҡзҡ„гҖҠSQL еӯ—е…ёгҖӢпјҢжүӢеҠЁжҠҠжҜҸеҸҘдёӯж–Үзҝ»иҜ‘жҲҗ SQLпјҢеҶҚжҠҠж•°жҚ®еә“иҝ”еӣһзҡ„з»“жһңжүӢеҠЁи§ЈжһҗжҲҗ Java еҜ№иұЎгҖӮжІЎжңүдёӯй—ҙе•Ҷиөҡе·®д»·пјҢдҪҶд№ҹжІЎжңүдёӯй—ҙе•Ҷеё®дҪ зңҒдәӢгҖӮ

38.1.2 JDBC зҡ„ж ёеҝғз»„д»¶

JDBC жңүеҮ дёӘдҪ еҝ…йЎ»и®°дҪҸзҡ„ж ёеҝғи§’иүІпјҡ

  • DriverManagerпјҡиҙҹиҙЈеҠ иҪҪж•°жҚ®еә“й©ұеҠЁпјҲжҜ”еҰӮ MySQL зҡ„й©ұеҠЁгҖҒOracle зҡ„й©ұеҠЁпјүпјҢе»әз«Ӣж•°жҚ®еә“иҝһжҺҘгҖӮ

import java.sql.*;
import java.util.ArrayList;
import java.util.List;

/**
 * JDBC CRUD е®Ңж•ҙжј”зӨә
 */
public class JdbcCrudDemo {

    private static final String URL = "jdbc:mysql://localhost:3306/myapp?useSSL=false&serverTimezone=UTC";
    private static final String USER = "root";
    private static final String PASSWORD = "123456";

    // ---------- C: жҸ’е…Ҙж•°жҚ® ----------
    public int insertUser(String username, String email) {
        String sql = "INSERT INTO users(username, email) VALUES(?, ?)";
        try (Connection conn = DriverManager.getConnection(URL, USER, PASSWORD);
             PreparedStatement ps = conn.prepareStatement(sql, Statement.RETURN_GENERATED_KEYS)) {
            ps.setString(1, username);
            ps.setString(2, email);
            int rows = ps.executeUpdate();
            if (rows > 0) {
                // иҺ·еҸ–иҮӘеўһдё»й”®
                try (ResultSet rs = ps.getGeneratedKeys()) {
                    if (rs.next()) {
                        return rs.getInt(1); // иҝ”еӣһж–°жҸ’е…Ҙи®°еҪ•зҡ„дё»й”®
                    }
                }
            }
        } catch (SQLException e) {
            System.err.println("жҸ’е…ҘеӨұиҙҘ: " + e.getMessage());
        }
        return -1;
    }

    // ---------- R: жҹҘиҜўж•°жҚ® ----------
    public List<String[]> queryAllUsers() {
        List<String[]> users = new ArrayList<>();
        String sql = "SELECT username, email FROM users";
        try (Connection conn = DriverManager.getConnection(URL, USER, PASSWORD);
             PreparedStatement ps = conn.prepareStatement(sql);
             ResultSet rs = ps.executeQuery()) {
            while (rs.next()) {
                users.add(new String[]{rs.getString("username"), rs.getString("email")});
            }
        } catch (SQLException e) {
            System.err.println("жҹҘиҜўеӨұиҙҘ: " + e.getMessage());
        }
        return users;
    }

    // ---------- U: жӣҙж–°ж•°жҚ® ----------
    public boolean updateUserEmail(int userId, String newEmail) {
        String sql = "UPDATE users SET email = ? WHERE id = ?";
        try (Connection conn = DriverManager.getConnection(URL, USER, PASSWORD);
             PreparedStatement ps = conn.prepareStatement(sql)) {
            ps.setString(1, newEmail);
            ps.setInt(2, userId);
            return ps.executeUpdate() > 0;
        } catch (SQLException e) {
            System.err.println("жӣҙж–°еӨұиҙҘ: " + e.getMessage());
            return false;
        }
    }

    // ---------- D: еҲ йҷӨж•°жҚ® ----------
    public boolean deleteUser(int userId) {
        String sql = "DELETE FROM users WHERE id = ?";
        try (Connection conn = DriverManager.getConnection(URL, USER, PASSWORD);
             PreparedStatement ps = conn.prepareStatement(sql)) {
            ps.setInt(1, userId);
            return ps.executeUpdate() > 0;
        } catch (SQLException e) {
            System.err.println("еҲ йҷӨеӨұиҙҘ: " + e.getMessage());
            return false;
        }
    }
}

 * з”ЁжҲ·е®һдҪ“зұ»
 * @Entity ж Үи®°дёә JPA е®һдҪ“
 * @Table жҢҮе®ҡеҜ№еә”зҡ„ж•°жҚ®еә“иЎЁеҗҚ

@Table(name = "users")
public class User {

    /**
     * @Id ж Үи®°дё»й”®
     * @GeneratedValue дё»й”®з”ҹжҲҗзӯ–з•Ҙ
     *    - IDENTITYпјҡиҮӘеўһй•ҝпјҲMySQLпјү
     *    - SEQUENCEпјҡеәҸеҲ—пјҲOracleгҖҒPostgreSQLпјү
     *    - TABLEпјҡиЎЁз”ҹжҲҗеҷЁпјҲйҖҡз”Ёпјү
     *    - AUTOпјҡз”ұ JPA иҮӘеҠЁйҖүжӢ©пјҲй»ҳи®Өпјү
     */
    @Id
    @GeneratedValue(strategy = GenerationType.IDENTITY)
    private Integer id;

    /**
     * @Column жҳ е°„еҲ—пјҢеҸҜжҢҮе®ҡеҲ—еҗҚгҖҒй•ҝеәҰгҖҒжҳҜеҗҰеҸҜз©ә
     */
    @Column(name = "username", nullable = false, length = 50)
    private String username;

    @Column(name = "email", length = 100)
    private String email;

    // ж— еҸӮжһ„йҖ еҷЁпјҲJPA еҝ…йЎ»пјү
    public User() {}

    public User(String username, String email) {
        this.username = username;
        this.email = email;
    }

    public Integer getId() { return id; }
    public void setId(Integer id) { this.id = id; }
    public String getUsername() { return username; }
    public void setUsername(String username) { this.username = username; }
    public String getEmail() { return email; }
    public void setEmail(String email) { this.email = email; }

    @Override
    public String toString() {
        return "User{id=" + id + ", username='" + username + "', email='" + email + "'}";
    }
}

import jakarta.persistence.*;
import java.util.List;

/**
 * JPA / Hibernate CRUD жј”зӨә
 */
public class JpaDemo {

    // еҲӣе»ә EntityManagerFactoryпјҲзӣёеҪ“дәҺ SqlSessionFactoryпјҢеҸҜеӨҚз”Ёпјү
    private static final EntityManagerFactory emf =
            Persistence.createEntityManagerFactory("myapp");

    public static void main(String[] args) {
        JpaDemo demo = new JpaDemo();

        // жҸ’е…Ҙ
        User user = demo.insertUser("zhangsan", "zhangsan@example.com");
        System.out.println("жҸ’е…Ҙ: " + user);

        // жҹҘиҜў
        User found = demo.findById(user.getId());
        System.out.println("жҹҘиҜў: " + found);

        // жӣҙж–°
        demo.updateEmail(user.getId(), "newemail@example.com");
        System.out.println("жӣҙж–°еҗҺжҹҘиҜў: " + demo.findById(user.getId()));

        // жҹҘиҜўжүҖжңү
        List<User> all = demo.findAll();
        all.forEach(System.out::println);

        // еҲ йҷӨ
        demo.deleteById(user.getId());
        System.out.println("еҲ йҷӨеҗҺжҹҘиҜў: " + demo.findById(user.getId()));

        // е…ій—ӯе·ҘеҺӮ
        emf.close();
    }

    // ---------- C: еҲӣе»ә ----------
    public User insertUser(String username, String email) {
        EntityManager em = emf.createEntityManager();
        EntityTransaction tx = em.getTransaction(); // JPA й»ҳи®ӨдёҚиҮӘеҠЁжҸҗдәӨ
        tx.begin();
        try {
            User user = new User(username, email);
            em.persist(user); // жҸ’е…ҘеҲ°ж•°жҚ®еә“
            tx.commit();
            return user;
        } catch (Exception e) {
            tx.rollback();
            throw e;
        } finally {
            em.close();
        }
    }

    // ---------- R: иҜ»еҸ– ----------
    public User findById(Integer id) {
        EntityManager em = emf.createEntityManager();
        try {
            // find ж–№жі•пјҡе…ҲжҹҘдёҖзә§зј“еӯҳпјҢжІЎжңүеҲҷжҹҘж•°жҚ®еә“
            return em.find(User.class, id);
        } finally {
            em.close();
        }
    }

    // дҪҝз”Ё JPQL жҹҘиҜўпјҲзұ»дјј SQLпјҢдҪҶж“ҚдҪңзҡ„жҳҜе®һдҪ“еҜ№иұЎпјү
    public List<User> findAll() {
        EntityManager em = emf.createEntityManager();
        try {
            TypedQuery<User> query = em.createQuery("SELECT u FROM User u", User.class);
            return query.getResultList();
        } finally {
            em.close();
        }
    }

    // ---------- U: жӣҙж–° ----------
    public void updateEmail(Integer id, String newEmail) {
        EntityManager em = emf.createEntityManager();
        EntityTransaction tx = em.getTransaction();
        tx.begin();
        try {
            User user = em.find(User.class, id);
            if (user != null) {
                user.setEmail(newEmail);
                em.merge(user); // жӣҙж–°пјҲmerge зұ»дјј saveOrUpdateпјү
            }
            tx.commit();
        } catch (Exception e) {
            tx.rollback();
            throw e;
        } finally {
            em.close();
        }
    }

    // ---------- D: еҲ йҷӨ ----------
    public void deleteById(Integer id) {
        EntityManager em = emf.createEntityManager();
        EntityTransaction tx = em.getTransaction();
        tx.begin();
        try {
            User user = em.find(User.class, id);
            if (user != null) {
                em.remove(user); // еҲ йҷӨ
            }
            tx.commit();
        } catch (Exception e) {
            tx.rollback();
            throw e;
        } finally {
            em.close();
        }
    }
}

38.3.7 JPA / Hibernate зҡ„дјҳзӮ№дёҺзјәзӮ№

дјҳзӮ№зјәзӮ№
е®Ңе…ЁеҜ№иұЎеҢ–пјҢдҪ еҸӘе…іеҝғ Java еҜ№иұЎеӨҚжқӮжҹҘиҜўпјҲеӨҡиЎЁ JOINпјүжҖ§иғҪдёҚеҰӮжүӢеҶҷ SQL
е‘ҠеҲ« SQLпјҢд»Јз Ғжӣҙз®ҖжҙҒпјҲе°Өе…¶жҳҜ Spring Data JPAпјүSQL дёҚйҖҸжҳҺпјҢи°ғдјҳеӣ°йҡҫпјҲ“й»‘зӣ’"й—®йўҳпјү
еӨ©з„¶ж”ҜжҢҒдёҖзә§зј“еӯҳпјҲsession зј“еӯҳпјүгҖҒдәҢзә§зј“еӯҳеӯҰд№ жӣІзәҝйҷЎеіӯпјҢжіЁи§Јдј—еӨҡ
ж•°жҚ®еә“з§»жӨҚжҖ§еҘҪпјҲжҚўж•°жҚ®еә“еҹәжң¬дёҚжҚўд»Јз ҒпјүжҮ’еҠ иҪҪеҜјиҮҙзҡ„ N+1 жҹҘиҜўй—®йўҳ
з”ҹжҖҒдё°еҜҢпјҲдёҺ Spring йӣҶжҲҗиүҜеҘҪпјүжү№йҮҸж“ҚдҪңжҖ§иғҪдёҚеҰӮ JDBC

жҖ»з»“пјҡJPA/Hibernate жҳҜйӮЈдёӘ"дҪ жғіеҒ·жҮ’пјҢе®ғеё®дҪ жҗһе®ҡдёҖеҲҮ"зҡ„йҖүжӢ©гҖӮдҪҶи®°дҪҸвҖ”вҖ”еҒ·жҮ’жҳҜжңүд»Јд»·зҡ„гҖӮеҪ“дҪ зҡ„ SQL еҸҳеҫ—еӨҚжқӮж—¶пјҢдҪ еҸҜиғҪдјҡиҠұжӣҙеӨҡж—¶й—ҙеҺ»зҗҶи§Ј Hibernate з”ҹжҲҗзҡ„ SQL еҲ°еә•еңЁе№Ід»Җд№ҲгҖӮ

жң¬з« е°Ҹз»“

жң¬з« жҲ‘д»¬д»Ӣз»ҚдәҶ Java з”ҹжҖҒдёӯдёүз§Қдё»жөҒзҡ„жҢҒд№…еҢ–ж–№жЎҲпјҢд»Һ"е…ЁжүӢеҠЁжҢЎ"еҲ°"е…ЁиҮӘеҠЁй©ҫй©¶”пјҡ

  1. JDBCпјҡJava ж•°жҚ®еә“йҖҡдҝЎзҡ„еҹәзҹігҖӮдҪ е®Ңе…ЁжҺҢжҺ§ SQLпјҢд№ҹе®Ңе…ЁиҙҹиҙЈжүҖжңүз»ҶиҠӮвҖ”вҖ”иҝһжҺҘз®ЎзҗҶгҖҒејӮеёёеӨ„зҗҶгҖҒз»“жһңйӣҶжҳ е°„гҖӮйҖӮеҗҲеҜ№жҖ§иғҪжңүжһҒиҮҙиҰҒжұӮгҖҒж„ҝж„ҸеҶҷеӨ§йҮҸжЁЎжқҝд»Јз Ғзҡ„еңәжҷҜгҖӮ

  2. MyBatisпјҡSQL жғіжҖҺд№ҲеҶҷдҪ иҜҙдәҶз®—пјҢз»“жһңйӣҶиҮӘеҠЁжҳ е°„пјҢеҠЁжҖҒ SQL ејәеӨ§гҖӮеӣҪеҶ…дә’иҒ”зҪ‘е…¬еҸёе® зҲұжңүеҠ пјҢеӣ дёәдёҡеҠЎеӨҚжқӮжҹҘиҜўеӨҡпјҢSQL дјҳеҢ–жҳҜеҲҡйңҖгҖӮдёҠжүӢз®ҖеҚ•пјҢиҝӣйҳ¶йңҖиҰҒзҗҶи§Је®ғзҡ„иҝҗиЎҢжңәеҲ¶гҖӮ

  3. JPA / HibernateпјҡеҜ№иұЎеҚідёҖеҲҮгҖӮдҪ ж“ҚдҪңеҜ№иұЎпјҢжЎҶжһ¶з”ҹжҲҗ SQLгҖӮSpring Data JPA жӣҙжҳҜжҠҠ"зәҰе®ҡдјҳдәҺй…ҚзҪ®"еҸ‘жҢҘеҲ°жһҒиҮҙгҖӮдҪҶеҪ“дҪ йңҖиҰҒеҶҷеӨҚжқӮжҹҘиҜўж—¶пјҢ“й»‘зӣ’"з”ҹжҲҗзҡ„ SQL еҸҜиғҪи®©дҪ жҠ“зӢӮгҖӮ

жІЎжңүжңҖеҘҪзҡ„жЎҶжһ¶пјҢеҸӘжңүжңҖйҖӮеҗҲзҡ„йҖүжӢ©гҖӮе°ҸйЎ№зӣ®еҝ«йҖҹејҖеҸ‘йҖү JPA/Spring Data JPAпјӣдёӯзӯүеӨҚжқӮеәҰйңҖиҰҒ SQL жҺ§еҲ¶еҠӣйҖү MyBatisпјӣжһҒиҮҙжҖ§иғҪе’Ңе®Ңе…ЁеҸҜжҺ§йҖү JDBCгҖӮеҪ“з„¶пјҢе®һйҷ…йЎ№зӣ®дёӯеҫҲеӨҡдәәжҳҜ"и„ҡиё©дёӨжқЎиҲ№”вҖ”вҖ”ж ёеҝғдёҡеҠЎз”Ё MyBatisпјҢз®ҖеҚ•зҡ„ CRUD з”Ё Spring Data JPAгҖӮ

жҢҒд№…еҢ–жҳҜеҗҺз«ҜејҖеҸ‘зҡ„еҹәзҹіпјҢеӯҰеҘҪиҝҷдёҖз« пјҢдҪ е’Ңж•°жҚ®еә“д№Ӣй—ҙзҡ„"еҜ№иҜқ"е°ұз®—жҳҜе…Ҙй—ЁдәҶпјҒ
